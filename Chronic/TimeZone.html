def self.scan(tokens, options)
  tokens.each do |token|
    if t = scan_for_all(token) then token.tag(t); next end
  end
end

def self.scan_for_all(token)
  scan_for token, self,
  {
    /[PMCE][DS]T/i => :tz,
    /(tzminus)?\d{4}/ => :tz
  }
end
