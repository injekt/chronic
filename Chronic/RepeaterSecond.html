def next(pointer = :future)
  super

  direction = pointer == :future ? 1 : -1

  if !@second_start
    @second_start = @now + (direction * SECOND_SECONDS)
  else
    @second_start += SECOND_SECONDS * direction
  end

  Span.new(@second_start, @second_start + SECOND_SECONDS)
end
